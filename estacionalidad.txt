function getColor(v) {
  const stops = [
    [1, '#e8f4f8'],
    [3, '#b8d8e8'],
    [5, '#7bb8d4'],
    [7, '#4a90c4'],
    [9, '#1a5fa0'],
    [10,'#0a3060']
  ];
  for (let i = stops.length - 1; i >= 0; i--) {
    if (v >= stops[i][0]) return stops[i][1];
  }
  return stops[0][1];
}

function getTextColor(v) {
  return v >= 6 ? '#ffffff' : 'var(--color-text-secondary)';
}

const tbody = document.getElementById('hm-body');
rubros.forEach(r => {
  const tr = document.createElement('tr');
  let html = `<td class="rubro-label">${r.name}</td>`;
  r.vals.forEach((v, i) => {
    const bg = getColor(v);
    const tc = getTextColor(v);
    const star = r.stars[i] ? '<sup style="font-size:9px;">★</sup>' : '';
    html += `<td><div class="cell" style="background:${bg};color:${tc};">${v}${star}</div></td>`;
  });
  tr.innerHTML = html;
  tbody.appendChild(tr);
});
